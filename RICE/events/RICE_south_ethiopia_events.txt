# Aari/South Omotic character spawn
south_ethiopia.0010 = {
	type = character_event
	title = south_ethiopia.0010.t
	desc = south_ethiopia.0010.desc
	theme = RICE_theme_south_ethiopia_omotic_village

	left_portrait = root
	right_portrait = {
		character = scope:courtier
		animation = throne_room_bow_1
	}

	trigger = {
		any_held_county = {
			any_county_province = {
				geographical_region = RICE_south_ethiopia_region
			}
		}
		NOR = {
			culture = culture:aari
			culture = {
				any_parent_culture_or_above = {
					this = culture:aari
				}
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2.5
			has_cultural_tradition = tradition_RICE_great_rift_valley
		}
	}

	immediate = {
		culture:aari = {
			save_scope_as = aari
		}
		create_character = {
			location = root.capital_province
			template = RICE_aari_generic_template
			save_scope_as = courtier			
		}
	}

	# Work with them
	option = {
		name = south_ethiopia.0010.a
		trigger = {
			OR = {
				gold >= 50
				is_ai = no
			}
		}
		remove_short_term_gold = 50
		add_courtier = scope:courtier
		if = {
			limit = {
				culture = { has_cultural_tradition = tradition_RICE_great_rift_valley }
			}
			capital_county = {
				add_county_modifier = {
					modifier = RICE_south_ethiopia_south_omotic_connections
					years = 20
				}
			}
		}
		else = {
			capital_county = {
				add_county_modifier = {
					modifier = RICE_south_ethiopia_south_omotic_connections
					years = 10
				}
			}
		}
	}
	
	# Don't work with them
	option = {
		name = south_ethiopia.0010.b
		hidden_effect = {
			random = {
				chance = 80
				scope:courtier = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}

}

			}
			triggered_desc = {
				trigger = {
					scope:host = {
						var:RICE_south_ethiopia_cattle_counting_speech = flag:good
					}
				}

			}
			triggered_desc = {
				trigger = {
					scope:host = {
						var:RICE_south_ethiopia_cattle_counting_speech = flag:excellent
					}
				}
				desc = south_ethiopia.0024.desc.excellent
			}
		}
	}
	theme = RICE_theme_south_ethiopia_landscape
	
	right_portrait = {
		character = root
		animation = personality_bold
	}
	
	option = { # Ok
		name = south_ethiopia.0024.a
		if = {
			limit = {
				var:RICE_south_ethiopia_cattle_counting_speech = flag:excellent
			}
			scope:activity = {
				add_activity_log_entry = {
					key = RICE_south_ethiopia_cattle_counting_ceremony_speech_excellent_log
					tags = { good }
					# this line below adds the entry to the Effects section of the conclusion UI
					show_in_conclusion = yes
					character = scope:host
					scope:host = {
						dynasty ?= {
							add_dynasty_prestige = RICE_cattle_counting_dynasty_prestige_excellent
						}
					}
				}
			}
		}
		else_if = {
			limit = {
				var:RICE_south_ethiopia_cattle_counting_speech = flag:good
			}
			scope:activity = {
				add_activity_log_entry = {
					key = RICE_south_ethiopia_cattle_counting_ceremony_speech_good_log
					tags = { good }
					# this line below adds the entry to the Effects section of the conclusion UI

					character = scope:host
					scope:host = {
						dynasty ?= {
							add_dynasty_prestige = RICE_cattle_counting_dynasty_prestige_good
						}
					}
				}
			}
		}
		else = {
			scope:activity = {
				add_activity_log_entry = {
					key = RICE_south_ethiopia_cattle_counting_ceremony_speech_fair_log

					character = scope:host
					scope:host = {
						dynasty ?= {
							add_dynasty_prestige = RICE_cattle_counting_dynasty_prestige_fair
						}

				}
			}
		}
	}

	after = {

			limit = { has_variable = RICE_south_ethiopia_cattle_counting_speech }
