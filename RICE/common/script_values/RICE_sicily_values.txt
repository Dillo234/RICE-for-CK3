
RICE_sicily_latin_stronghold_appearance_value = {
	value = 5
	if = {
		limit = {
			NOT = {
				faith.religious_head_title ?= title:k_papal_state
			}
		}
		multiply = 1.5
	}
	if = {
		limit = {
			NOR = {
				culture = { has_cultural_pillar = heritage_latin }
				culture = { has_cultural_pillar = heritage_frankish }
				culture = { has_cultural_pillar = heritage_iberian }
			}
		}
		multiply = 1.5
	}
	if = {
		limit = {
			any_held_county = {
				faith.religious_head_title ?= title:k_papal_state
			}
		}
		multiply = 1.5
	}
	if = {
		limit = {
			any_held_county = {
				OR = {
					culture = { has_cultural_pillar = heritage_latin }
					culture = { has_cultural_pillar = heritage_frankish }
					culture = { has_cultural_pillar = heritage_iberian }
				}
			}
		}
		multiply = 1.5
	}
	min = 1
	max = 30
}

RICE_sicily_latin_influence_global_value = {
	value = 0
	struggle:RICE_sicily_struggle ?= {
		every_involved_ruler = {
			limit = {
				highest_held_title_tier >= tier_county
				OR = {
					faith.religious_head_title ?= title:k_papal_state
					culture = { has_cultural_pillar = heritage_latin }
					culture = { has_cultural_pillar = heritage_frankish }
					culture = { has_cultural_pillar = heritage_iberian }
				}
			}
			add = 5
		}
		every_interloper_ruler = {
			limit = {
				highest_held_title_tier >= tier_county
				OR = {
					faith.religious_head_title ?= title:k_papal_state
					culture = { has_cultural_pillar = heritage_latin }
					culture = { has_cultural_pillar = heritage_frankish }
					culture = { has_cultural_pillar = heritage_iberian }
				}
			}
			add = 3
		}
		every_involved_ruler = {
			limit = {
				highest_held_title_tier >= tier_county
				has_trait_xp = {
					trait = RICE_sicilian_ruler
					track = RICE_sicilian_ruler_latin
					value >= trait_second_level
				}
			}
			add = 2
		}
		every_interloper_ruler = {
			limit = {
				highest_held_title_tier >= tier_county
				has_trait_xp = {
					trait = RICE_sicilian_ruler
					track = RICE_sicilian_ruler_latin
					value >= trait_second_level
				}
			}
			add = 1
		}
		every_involved_ruler = {
			limit = {
				highest_held_title_tier >= tier_county
				has_trait_xp = {
					trait = RICE_sicilian_ruler
					track = RICE_sicilian_ruler_latin
					value >= trait_third_level
				}
			}
			add = 3

					trait = RICE_sicilian_ruler
					track = RICE_sicilian_ruler_latin

			}
			add = 2
		}
		every_involved_county = {

					culture = { has_cultural_pillar = heritage_latin }
