RICE_india_grant_decision_cost = {
	value = 25
	every_held_title = {
		limit = {
			tier = tier_county
		}
		add = 25
	}
	if = {
		limit = {
			root.primary_title.tier <= tier_duchy
		}
		multiply = 0.8
	}
	# Make it a little easier for smaller realms
	if = {
		limit = {
			root= {
				sub_realm_size <= 4
			}
		}
		multiply = 0.8
	}
	min = 25
	max = 300
}
