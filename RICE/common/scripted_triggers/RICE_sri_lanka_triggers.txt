		AND = {
			any_held_county = {
				any_county_province = {
					geographical_region = RICE_sri_lanka_struggle_region
				}
			}
			OR = {
				religion = religion:buddhism_religion
				faith = { has_doctrine = doctrine_pluralism_pluralistic }
				#faith = { has_doctrine = tenet_eastern_syncretism }
			}	
		}

	}
}
