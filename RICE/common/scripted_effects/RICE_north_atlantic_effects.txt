# Add all Involved cultures with counties to a list
RICE_greenland_struggle_ending_culture_list_effect = {
	every_culture_global = {		
		limit = {
			#TODO_CD_FP2 (JP) check if this can be more performant, every_struggle_culture would be better
			save_temporary_scope_as = this_culture
			struggle:RICE_greenland_struggle = { is_culture_involved_in_struggle = scope:this_culture }
			culture_number_of_counties > 0
		}
		save_scope_as = culture_scope
		add_to_global_variable_list = {
			name = RICE_greenland_struggle_ending_culture_list
			target = scope:culture_scope
		}
	}
}



# Add cultural acceptance between Involved cultures
RICE_greenland_struggle_integration_cultural_acceptance_effect = {
	hidden_effect = {
		while = {
			limit = {
				any_in_global_list = {
					variable = RICE_greenland_struggle_ending_culture_list
					NOT = { has_variable = RICE_greenland_struggle_integration_flag }
				}
			}
			ordered_in_global_list = {
				variable = RICE_greenland_struggle_ending_culture_list
				limit = {
					NOT = { has_variable = RICE_greenland_struggle_integration_flag }
				}
				order_by = culture_number_of_counties
				save_scope_as = culture_scope
				set_variable = RICE_greenland_struggle_integration_flag # variable ensures acceptance is only gained once per couple
			}
			every_in_global_list = {
				variable = RICE_greenland_struggle_ending_culture_list
				limit = {
					NOR = {
						has_variable = RICE_greenland_struggle_integration_flag
						this = scope:culture_scope
					}
				}
				change_cultural_acceptance = {
					target = scope:culture_scope
					value = fp2_struggle_conciliation_culture_acceptance_value #TODO_CD_FP2 (JP) check if this is enough
					desc = RICE_greenland_struggle_integration_acceptance_tt
				}
			}
		}
		every_in_global_list = {
			variable = RICE_greenland_struggle_ending_culture_list
			remove_variable = RICE_greenland_struggle_integration_flag
		}
	}
}
